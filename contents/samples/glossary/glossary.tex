\chapter{Use of Glossary}

\gls{computer} % To print the term, lowercase. For example, \gls{maths} prints mathematics when used.

\Gls{computer} % The same as \gls but the first letter will be printed in uppercase. Example: \Gls{maths} prints Mathematics

\glspl{computer} % The same as \gls but the term is put in its plural form. For instance, \glspl{formula} will write formulas in your final document.

\Glspl{computer} % The same as \Gls but the term is put in its plural form. For example, \Glspl{formula} renders as Formulas.
