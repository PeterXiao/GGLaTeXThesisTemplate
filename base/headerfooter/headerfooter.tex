\fancyhf{}

% E for even page
% O for odd page
% L for left side
% C for centered
% R for right side

% set header
%--------------------------------------

\fancyhead[R]{\thepage}
\fancyhead[C]{}
%--------------------------------------

% change header line color
\patchcmd{\headrule}{\hrule}{\color{LightGray}\hrule}{}{}
% show header line
\renewcommand{\headrulewidth}{0.4pt} % set to 0pt if line shall be removed

% set footer
%--------------------------------------
\fancyfoot[R]{}
\fancyfoot[L]{}
\fancyfoot[C]{}
%--------------------------------------
